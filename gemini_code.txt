    <script>
        // --- Configuração de Clusters e KG ---
        const CLUSTERS = [
            { id: 'ai', name: 'Machine Learning', color: 0x00A3FF, center: [0, 0, 0], scale: 1.2 },
            { id: 'bio', name: 'Bio-Engineering', color: 0xFF0055, center: [100, 40, -80], scale: 0.9 },
            { id: 'quantum', name: 'Quantum Computing', color: 0xFFB800, center: [-110, -50, 60], scale: 1.0 },
            { id: 'neuro', name: 'Neuroscience', color: 0x00FF85, center: [50, -90, -40], scale: 1.1 }
        ];

        let papersData = [];
        const TOTAL_POINTS = 4000;

        function generateData() {
            for(let i=0; i<TOTAL_POINTS; i++) {
                const cluster = CLUSTERS[Math.floor(Math.random() * CLUSTERS.length)];
                
                // Distribuição tipo Nuvem de Probabilidade (mais orgânico)
                const std = 25 + Math.random() * 25;
                const x = cluster.center[0] + (Math.random() - 0.5) * std * 4;
                const y = cluster.center[1] + (Math.random() - 0.5) * std * 3;
                const z = cluster.center[2] + (Math.random() - 0.5) * std * 4;

                const centrality = Math.random();
                const connections = Math.floor(Math.random() * 30);

                papersData.push({
                    id: i,
                    topic: cluster.name,
                    topicColor: cluster.color,
                    centrality: centrality,
                    connections: connections,
                    x: x, y: y, z: z,
                    title: `Investigação Relacional v.${i} em ${cluster.name}`,
                    abstract: "O presente estudo utiliza grafos de conhecimento para mapear a evolução de conceitos interdisciplinares, revelando que as fronteiras entre domínios são onde ocorre a inovação radical..."
                });
            }
        }
        generateData();

        // --- Motor Three.js ---
        let scene, camera, renderer, controls, raycaster, mouse, points, geometry, material;
        let lineMesh = null;
        let nebulaParticles = [];
        let selectedIndex = -1, hoveredIndex = -1;

        function init() {
            scene = new THREE.Scene();
            scene.fog = new THREE.FogExp2(0x010103, 0.003);

            camera = new THREE.PerspectiveCamera(50, window.innerWidth / window.innerHeight, 0.1, 2000);
            camera.position.set(0, 200, 500);

            renderer = new THREE.WebGLRenderer({ antialias: true, alpha: true });
            renderer.setSize(window.innerWidth, window.innerHeight);
            renderer.setPixelRatio(Math.min(window.devicePixelRatio, 2));
            document.getElementById('canvas-container').appendChild(renderer.domElement);

            controls = new THREE.OrbitControls(camera, renderer.domElement);
            controls.enableDamping = true;
            controls.autoRotate = true;
            controls.autoRotateSpeed = 0.12;

            // 1. Adicionar Nebulosas (Cluster Shrouds)
            createNebulae();

            // 2. Criar a Galáxia de Pontos
            createGalaxy();

            animate();

            window.addEventListener('resize', onWindowResize);
            document.addEventListener('mousemove', onMouseMove);
            document.addEventListener('click', onClick);
        }

        function createNebulae() {
            CLUSTERS.forEach(cluster => {
                const nebulaGeo = new THREE.SphereGeometry(60 * cluster.scale, 32, 32);
                const nebulaMat = new THREE.MeshBasicMaterial({
                    color: cluster.color,
                    transparent: true,
                    opacity: 0.03,
                    blending: THREE.AdditiveBlending,
                    side: THREE.BackSide
                });
                const nebula = new THREE.Mesh(nebulaGeo, nebulaMat);
                nebula.position.set(...cluster.center);
                scene.add(nebula);
                
                // Adicionar um halo mais brilhante no centro
                const haloGeo = new THREE.SphereGeometry(10, 16, 16);
                const haloMat = new THREE.MeshBasicMaterial({ color: cluster.color, transparent: true, opacity: 0.1 });
                const halo = new THREE.Mesh(haloGeo, haloMat);
                halo.position.set(...cluster.center);
                scene.add(halo);
            });
        }

        function createGalaxy() {
            geometry = new THREE.BufferGeometry();
            const pos = new Float32Array(TOTAL_POINTS * 3);
            const col = new Float32Array(TOTAL_POINTS * 3);
            const siz = new Float32Array(TOTAL_POINTS);

            const c = new THREE.Color();
            papersData.forEach((p, i) => {
                pos[i*3] = p.x; pos[i*3+1] = p.y; pos[i*3+2] = p.z;
                c.setHex(p.topicColor);
                col[i*3] = c.r; col[i*3+1] = c.g; col[i*3+2] = c.b;
                // Tamanho baseado na centralidade (Hubs são maiores)
                siz[i] = 1.5 + (p.centrality * 4.5);
            });

            geometry.setAttribute('position', new THREE.BufferAttribute(pos, 3));
            geometry.setAttribute('color', new THREE.BufferAttribute(col, 3));
            geometry.setAttribute('size', new THREE.BufferAttribute(siz, 1));

            material = new THREE.ShaderMaterial({
                vertexShader: `
                    attribute float size;
                    attribute vec3 color;
                    varying vec3 vColor;
                    void main() {
                        vColor = color;
                        vec4 mvPos = modelViewMatrix * vec4(position, 1.0);
                        gl_PointSize = size * (450.0 / -mvPos.z);
                        gl_Position = projectionMatrix * mvPos;
                    }
                `,
                fragmentShader: `
                    varying vec3 vColor;
                    void main() {
                        float dist = length(gl_PointCoord - vec2(0.5));
                        if(dist > 0.5) discard;
                        float core = smoothstep(0.4, 0.0, dist * 3.0);
                        float glow = pow(1.0 - dist*2.0, 1.8);
                        gl_FragColor = vec4(vColor * (1.1 + core * 5.0), glow);
                    }
                `,
                transparent: true,
                blending: THREE.AdditiveBlending,
                depthWrite: false
            });

            points = new THREE.Points(geometry, material);
            scene.add(points);
        }

        function onMouseMove(e) {
            mouse = new THREE.Vector2((e.clientX / window.innerWidth) * 2 - 1, -(e.clientY / window.innerHeight) * 2 + 1);
            raycaster = new THREE.Raycaster();
            raycaster.setFromCamera(mouse, camera);
            raycaster.params.Points.threshold = 1.5;
            const intersects = raycaster.intersectObject(points);
            const tt = document.getElementById('tooltip');

            if(intersects.length > 0) {
                const idx = intersects[0].index;
                if(hoveredIndex !== idx) {
                    hoveredIndex = idx;
                    const p = papersData[idx];
                    document.getElementById('tt-title').innerText = p.title;
                    document.getElementById('tt-topic').innerText = p.topic;
                    document.getElementById('tt-connections').innerText = `${p.connections} Adjacências`;
                    document.getElementById('tt-id').innerText = `KG: ${p.centrality.toFixed(2)}`;
                    tt.style.opacity = '1';
                    tt.style.transform = 'translateY(0) scale(1)';
                }
                tt.style.left = (e.clientX + 25) + 'px';
                tt.style.top = (e.clientY + 25) + 'px';
            } else {
                hoveredIndex = -1;
                tt.style.opacity = '0';
                tt.style.transform = 'translateY(15px) scale(0.9)';
            }
        }

        function onClick() {
            if(!mouse) return;
            raycaster.setFromCamera(mouse, camera);
            const intersects = raycaster.intersectObject(points);
            if(intersects.length > 0) {
                const idx = intersects[0].index;
                selectedIndex = idx;
                const p = papersData[idx];
                
                showWorkpad(p);
                
                // Voo de Câmara
                gsap.to(camera.position, { duration: 2, x: p.x, y: p.y + 15, z: p.z + 70, ease: "expo.out" });
                gsap.to(controls.target, { duration: 2, x: p.x, y: p.y, z: p.z, ease: "expo.out" });
                controls.autoRotate = false;

                // Desenhar Vizinhos Imediatos (Citações reais)
                drawConnections(idx);
            }
        }

        function drawConnections(centerIdx) {
            if(lineMesh) scene.remove(lineMesh);
            
            const center = papersData[centerIdx];
            const positions = [];
            const colors = [];
            const color = new THREE.Color(center.topicColor);

            // Simular 8 vizinhos mais próximos no mesmo cluster
            for(let i=0; i<8; i++) {
                const neighborIdx = (centerIdx + i + 1) % TOTAL_POINTS;
                const neighbor = papersData[neighborIdx];
                positions.push(center.x, center.y, center.z);
                positions.push(neighbor.x, neighbor.y, neighbor.z);
                colors.push(color.r, color.g, color.b);
                colors.push(color.r, color.g, color.b);
            }

            const lineGeo = new THREE.BufferGeometry();
            lineGeo.setAttribute('position', new THREE.Float32BufferAttribute(positions, 3));
            lineGeo.setAttribute('color', new THREE.Float32BufferAttribute(colors, 3));
            
            const lineMat = new THREE.LineBasicMaterial({ 
                vertexColors: true, 
                transparent: true, 
                opacity: 0.2, 
                blending: THREE.AdditiveBlending 
            });
            lineMesh = new THREE.LineSegments(lineGeo, lineMat);
            scene.add(lineMesh);
        }

        function visualizeHiddenRelationships() {
            const insight = document.getElementById('kg-insight');
            insight.classList.remove('hidden');
            setTimeout(() => insight.classList.remove('opacity-0', 'translate-y-6'), 100);

            // Criar uma ponte de luz pulsante para outro cluster
            const p1 = papersData[selectedIndex];
            const targetCluster = CLUSTERS.find(c => c.name !== p1.topic);
            // Encontrar um hub no cluster alvo
            const p2 = papersData.find(p => p.topic === targetCluster.name && p.centrality > 0.8);

            const curve = new THREE.CatmullRomCurve3([
                new THREE.Vector3(p1.x, p1.y, p1.z),
                new THREE.Vector3((p1.x+p2.x)/2, (p1.y+p2.y)/2 + 60, (p1.z+p2.z)/2),
                new THREE.Vector3(p2.x, p2.y, p2.z)
            ]);

            const tubeGeo = new THREE.TubeGeometry(curve, 100, 0.25, 8, false);
            const tubeMat = new THREE.MeshBasicMaterial({ color: 0x3b82f6, transparent: true, opacity: 0.7, blending: THREE.AdditiveBlending });
            const bridge = new THREE.Mesh(tubeGeo, tubeMat);
            scene.add(bridge);

            // Animação de "Disparo" de informação
            tubeGeo.setDrawRange(0, 0);
            let draw = { v: 0 };
            gsap.to(draw, { v: 1, duration: 2, ease: "power2.inOut", onUpdate: () => tubeGeo.setDrawRange(0, Math.floor(draw.v * 10000)) });
        }

        function updateView(view) {
            document.querySelectorAll('.lens-btn').forEach(b => {
                if(b.dataset.view === view) b.classList.add('active', 'bg-white/10', 'text-white');
                else b.classList.remove('active', 'bg-white/10', 'text-white');
            });

            const colAttr = geometry.attributes.color;
            const sizeAttr = geometry.attributes.size;
            const c = new THREE.Color();

            papersData.forEach((p, i) => {
                if(view === 'centrality') {
                    // Visão de Calor: Vermelho (Hubs) para Azul (Periferia)
                    c.setHSL(0.66 * (1 - p.centrality), 1.0, 0.5);
                    sizeAttr.array[i] = 1.0 + (p.centrality * 8.0);
                } else if(view === 'hidden') {
                    // Escurece tudo exceto os nós com alto potencial de Gap
                    const isGap = p.centrality > 0.4 && p.centrality < 0.6;
                    c.setHex(isGap ? 0xfbb03b : 0x111111);
                    sizeAttr.array[i] = isGap ? 6.0 : 1.0;
                } else {
                    c.setHex(p.topicColor);
                    sizeAttr.array[i] = 1.5 + (p.centrality * 4.5);
                }
                colAttr.array[i*3] = c.r; colAttr.array[i*3+1] = c.g; colAttr.array[i*3+2] = c.b;
            });
            colAttr.needsUpdate = true;
            sizeAttr.needsUpdate = true;
        }

        function showWorkpad(p) {
            const wp = document.getElementById('workpad');
            document.getElementById('wp-title').innerText = p.title;
            document.getElementById('wp-topic').innerText = p.topic;
            document.getElementById('wp-conn-count').innerText = p.connections;
            document.getElementById('kg-insight').classList.add('hidden', 'opacity-0', 'translate-y-6');
            wp.classList.remove('translate-x-full');
        }

        function closeWorkpad() {
            document.getElementById('workpad').classList.add('translate-x-full');
            controls.autoRotate = true;
            if(lineMesh) scene.remove(lineMesh);
        }

        function onWindowResize() {
            camera.aspect = window.innerWidth / window.innerHeight;
            camera.updateProjectionMatrix();
            renderer.setSize(window.innerWidth, window.innerHeight);
        }

        function animate() {
            requestAnimationFrame(animate);
            controls.update();
            renderer.render(scene, camera);
        }

        window.onload = init;

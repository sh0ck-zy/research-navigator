    <script>
        // --- Dados ---
        const CLUSTERS = [
            { id: 'ml', name: 'Machine Learning', color: 0x8AB4F8, center: [0, 0, 0] },
            { id: 'neuro', name: 'Neuroscience', color: 0xF28B82, center: [120, 40, -60] },
            { id: 'bio', name: 'Bioinformatics', color: 0x81C995, center: [-100, -30, 80] },
            { id: 'quantum', name: 'Quantum Computing', color: 0xFDE293, center: [50, -100, -50] }
        ];

        let papersData = [];
        const TOTAL_POINTS = 4000;

        function generateData() {
            for(let i=0; i<TOTAL_POINTS; i++) {
                const cluster = CLUSTERS[Math.floor(Math.random() * CLUSTERS.length)];
                
                // Distribuição Gaussiana apertada para criar "ilhas" claras
                const u = 1 - Math.random(); 
                const v = Math.random();
                const std = 30;
                const radius = std * Math.sqrt(-2.0 * Math.log(u));
                const theta = 2.0 * Math.PI * v;
                
                const x = cluster.center[0] + radius * Math.cos(theta);
                const y = cluster.center[1] + radius * Math.sin(theta);
                const z = cluster.center[2] + (Math.random() - 0.5) * std * 2;

                papersData.push({
                    id: i,
                    topic: cluster.name,
                    topicColor: cluster.color,
                    topicColorHex: '#' + cluster.color.toString(16).padStart(6, '0'),
                    citations: Math.floor(Math.random() * 5000),
                    year: 2015 + Math.floor(Math.random() * 11),
                    trl: Math.floor(Math.random() * 9) + 1,
                    x: x, y: y, z: z,
                    title: `Estudo Exploratório em ${cluster.name} #${i}`,
                    abstract: "Nesta investigação, projetamos os embeddings de linguagem num espaço de alta dimensão, permitindo descobrir relações latentes através de modelos de similaridade de cosseno..."
                });
            }
        }
        generateData();

        // --- Motor 3D ---
        let scene, camera, renderer, controls, raycaster, mouse, points, geometry, material;
        let connectionsLines = null;
        let opacities;

        function init() {
            scene = new THREE.Scene();
            scene.fog = new THREE.FogExp2(0x121212, 0.0025);

            camera = new THREE.PerspectiveCamera(45, window.innerWidth / window.innerHeight, 0.1, 2000);
            camera.position.set(0, 150, 450);

            renderer = new THREE.WebGLRenderer({ antialias: true, alpha: false });
            renderer.setSize(window.innerWidth, window.innerHeight);
            renderer.setPixelRatio(Math.min(window.devicePixelRatio, 2));
            renderer.setClearColor(0x121212);
            document.getElementById('canvas-container').appendChild(renderer.domElement);

            controls = new THREE.OrbitControls(camera, renderer.domElement);
            controls.enableDamping = true;
            controls.dampingFactor = 0.05;
            controls.autoRotate = true;
            controls.autoRotateSpeed = 0.5;

            createGalaxy();
            animate();

            window.addEventListener('resize', onWindowResize);
            document.addEventListener('mousemove', onMouseMove);
            document.addEventListener('click', onClick);
        }

        function createGalaxy() {
            geometry = new THREE.BufferGeometry();
            const pos = new Float32Array(TOTAL_POINTS * 3);
            const col = new Float32Array(TOTAL_POINTS * 3);
            const siz = new Float32Array(TOTAL_POINTS);
            opacities = new Float32Array(TOTAL_POINTS); // Atributo crucial para o Search Filter

            const c = new THREE.Color();
            papersData.forEach((p, i) => {
                pos[i*3] = p.x; pos[i*3+1] = p.y; pos[i*3+2] = p.z;
                c.setHex(p.topicColor);
                col[i*3] = c.r; col[i*3+1] = c.g; col[i*3+2] = c.b;
                siz[i] = 2.0;
                opacities[i] = 1.0; // Tudo visível por defeito
            });

            geometry.setAttribute('position', new THREE.BufferAttribute(pos, 3));
            geometry.setAttribute('color', new THREE.BufferAttribute(col, 3));
            geometry.setAttribute('size', new THREE.BufferAttribute(siz, 1));
            geometry.setAttribute('opacity', new THREE.BufferAttribute(opacities, 1));

            // Shader Minimalista (Pontos nítidos sem glow excessivo)
            material = new THREE.ShaderMaterial({
                vertexShader: `
                    attribute float size;
                    attribute vec3 color;
                    attribute float opacity;
                    varying vec3 vColor;
                    varying float vOpacity;
                    void main() {
                        vColor = color;
                        vOpacity = opacity;
                        vec4 mvPos = modelViewMatrix * vec4(position, 1.0);
                        gl_PointSize = size * (300.0 / -mvPos.z);
                        gl_Position = projectionMatrix * mvPos;
                    }
                `,
                fragmentShader: `
                    varying vec3 vColor;
                    varying float vOpacity;
                    void main() {
                        float dist = length(gl_PointCoord - vec2(0.5));
                        if(dist > 0.5) discard;
                        // Círculo sólido suave
                        float alpha = smoothstep(0.5, 0.4, dist) * vOpacity;
                        gl_FragColor = vec4(vColor, alpha);
                    }
                `,
                transparent: true,
                depthWrite: false,
                blending: THREE.NormalBlending // Normal blending em vez de Additive para look mais limpo
            });

            points = new THREE.Points(geometry, material);
            scene.add(points);
        }

        // --- Lógica de Pesquisa e Filtros ---
        function handleSearch(term) {
            term = term.toLowerCase();
            const opacityAttr = geometry.attributes.opacity;
            
            papersData.forEach((p, i) => {
                // Se o termo estiver no título ou tópico, opacidade 1.0, senão 0.1
                const match = p.title.toLowerCase().includes(term) || p.topic.toLowerCase().includes(term);
                opacityAttr.array[i] = (term === "" || match) ? 1.0 : 0.05;
            });
            opacityAttr.needsUpdate = true;
        }

        function filterByTopic(topic, element) {
            // Atualizar UI dos Chips
            document.querySelectorAll('.chip').forEach(c => c.classList.remove('active'));
            element.classList.add('active');

            const opacityAttr = geometry.attributes.opacity;
            papersData.forEach((p, i) => {
                const match = topic === 'All' || p.topic === topic;
                // Animação suave de opacidade via GSAP para cada ponto não é perfomante, usamos shader update
                opacityAttr.array[i] = match ? 1.0 : 0.05;
            });
            opacityAttr.needsUpdate = true;
            
            // Focar a câmara no cluster se não for 'All'
            if(topic !== 'All') {
                const cluster = CLUSTERS.find(c => c.name === topic);
                if(cluster) {
                    gsap.to(camera.position, { duration: 1.5, x: cluster.center[0], y: cluster.center[1] + 100, z: cluster.center[2] + 200, ease: "power2.out" });
                    gsap.to(controls.target, { duration: 1.5, x: cluster.center[0], y: cluster.center[1], z: cluster.center[2], ease: "power2.out" });
                }
            } else {
                controls.autoRotate = true;
            }
        }

        // --- Interações ---
        function onMouseMove(e) {
            mouse = new THREE.Vector2((e.clientX / window.innerWidth) * 2 - 1, -(e.clientY / window.innerHeight) * 2 + 1);
            raycaster = new THREE.Raycaster();
            raycaster.setFromCamera(mouse, camera);
            raycaster.params.Points.threshold = 1.5;
            
            // Só interagir com pontos visíveis
            const intersects = raycaster.intersectObject(points).filter(i => geometry.attributes.opacity.array[i.index] > 0.5);
            
            const tt = document.getElementById('tooltip');

            if(intersects.length > 0) {
                const p = papersData[intersects[0].index];
                document.getElementById('tt-title').innerText = p.title;
                document.getElementById('tt-topic').innerText = p.topic;
                document.getElementById('tt-topic').style.color = p.topicColorHex;
                document.getElementById('tt-meta').innerText = `Citações: ${p.citations}`;
                
                tt.style.opacity = '1';
                tt.style.left = (e.clientX + 15) + 'px';
                tt.style.top = (e.clientY + 15) + 'px';
                document.body.style.cursor = 'pointer';
            } else {
                tt.style.opacity = '0';
                document.body.style.cursor = 'default';
            }
        }

        function onClick() {
            if(!mouse) return;
            raycaster.setFromCamera(mouse, camera);
            const intersects = raycaster.intersectObject(points).filter(i => geometry.attributes.opacity.array[i.index] > 0.5);
            
            if(intersects.length > 0) {
                const idx = intersects[0].index;
                const p = papersData[idx];
                
                openSidePanel(p, idx);
                drawNearestNeighbors(idx);
                
                // Fly to point
                gsap.to(camera.position, { duration: 1.5, x: p.x, y: p.y + 20, z: p.z + 60, ease: "power3.out" });
                gsap.to(controls.target, { duration: 1.5, x: p.x, y: p.y, z: p.z, ease: "power3.out" });
                controls.autoRotate = false;
            }
        }

        function openSidePanel(p, idx) {
            const wp = document.getElementById('workpad');
            document.getElementById('wp-title').innerText = p.title;
            document.getElementById('wp-topic').innerText = p.topic;
            document.getElementById('wp-color-dot').style.backgroundColor = p.topicColorHex;
            document.getElementById('wp-doi').innerText = `DOI: 10.1038/${p.id}`;
            document.getElementById('wp-citations').innerText = p.citations;
            document.getElementById('wp-year').innerText = p.year;
            document.getElementById('wp-trl').innerText = `TRL ${p.trl}`;
            
            generateNeighborsList(p, idx);
            
            wp.classList.remove('translate-x-full');
        }

        function closeWorkpad() {
            document.getElementById('workpad').classList.add('translate-x-full');
            if(connectionsLines) scene.remove(connectionsLines);
            controls.autoRotate = true;
        }

        // --- Motor de Relacionamento (Nearest Neighbors) ---
        function getNearestNeighbors(centerIdx, count = 5) {
            const center = papersData[centerIdx];
            const distances = papersData.map((p, idx) => {
                if(idx === centerIdx) return { idx, dist: Infinity };
                // Distância Euclidiana 3D simulando proximidade de embeddings
                const dist = Math.sqrt(Math.pow(p.x - center.x, 2) + Math.pow(p.y - center.y, 2) + Math.pow(p.z - center.z, 2));
                return { idx, dist };
            });
            distances.sort((a, b) => a.dist - b.dist);
            return distances.slice(0, count).map(d => papersData[d.idx]);
        }

        function drawNearestNeighbors(centerIdx) {
            if(connectionsLines) scene.remove(connectionsLines);
            
            const center = papersData[centerIdx];
            const neighbors = getNearestNeighbors(centerIdx, 5);
            
            const positions = [];
            const colors = [];
            const centerColor = new THREE.Color(center.topicColor);

            neighbors.forEach(n => {
                positions.push(center.x, center.y, center.z);
                positions.push(n.x, n.y, n.z);
                
                const neighborColor = new THREE.Color(n.topicColor);
                colors.push(centerColor.r, centerColor.g, centerColor.b);
                colors.push(neighborColor.r, neighborColor.g, neighborColor.b); // Gradiente na linha
            });

            const geo = new THREE.BufferGeometry();
            geo.setAttribute('position', new THREE.Float32BufferAttribute(positions, 3));
            geo.setAttribute('color', new THREE.Float32BufferAttribute(colors, 3));
            
            const mat = new THREE.LineBasicMaterial({ vertexColors: true, transparent: true, opacity: 0.6 });
            connectionsLines = new THREE.LineSegments(geo, mat);
            scene.add(connectionsLines);
        }

        function generateNeighborsList(centerPaper, centerIdx) {
            const neighbors = getNearestNeighbors(centerIdx, 5);
            const container = document.getElementById('nearest-neighbors-list');
            container.innerHTML = '';

            neighbors.forEach((n, i) => {
                const matchPercentage = (99 - (i * 4.2)).toFixed(1); // Simulação de score semântico
                const div = document.createElement('div');
                div.className = 'p-3 rounded-lg bg-surface border border-border hover:bg-[#2A2A2A] cursor-pointer transition-colors flex items-center space-x-3';
                div.innerHTML = `
                    <div class="w-8 h-8 rounded bg-[#121212] flex items-center justify-center flex-shrink-0 border border-border">
                        <span class="text-xs font-bold" style="color: ${n.topicColorHex}">${matchPercentage}%</span>
                    </div>
                    <div class="overflow-hidden">
                        <p class="text-xs font-medium text-white truncate">${n.title}</p>
                        <p class="text-[10px] text-gray-500 truncate">${n.topic} • ${n.year}</p>
                    </div>
                `;
                // Fly to this neighbor when clicked
                div.onclick = () => {
                    const idx = papersData.indexOf(n);
                    openSidePanel(n, idx);
                    drawNearestNeighbors(idx);
                    gsap.to(camera.position, { duration: 1.5, x: n.x, y: n.y + 20, z: n.z + 60, ease: "power3.out" });
                    gsap.to(controls.target, { duration: 1.5, x: n.x, y: n.y, z: n.z });
                };
                container.appendChild(div);
            });
        }

        // --- Render Loop ---
        function onWindowResize() {
            camera.aspect = window.innerWidth / window.innerHeight;
            camera.updateProjectionMatrix();
            renderer.setSize(window.innerWidth, window.innerHeight);
        }

        function animate() {
            requestAnimationFrame(animate);
            controls.update();
            renderer.render(scene, camera);
        }

        window.onload = init;
    </script>
